	async _deleteTenantFolder(ctxt) {
		try {
			const dbSrvc = this.$dependencies.DatabaseService;
			const ApiService = this.$dependencies.ApiService;

			const tenantFolder = await this.$TenantFolderModel
			.query(function(qb) {
				qb
				.where({ 'id': ctxt.params['folder_id'] })
				.andWhere({ 'tenant_id': ctxt.state.tenant['tenant_id'] });
			})
			.fetch();

			if(!tenantFolder) throw new Error('Unknown Tenant Folder');

			const cacheMulti = this.$dependencies.CacheService.multi();
			const cachedFolderArtifacts = await ApiService.execute('DataProcessorManager::_getCachedFolderArtifacts', ctxt.params['folder_id']);

			const folderDescendants = await dbSrvc.knex.raw(`SELECT id FROM fn_get_folder_descendants(?);`, [ctxt.params['folder_id']]);

			// eslint-disable-next-line
			const folderPanelConstituents = await dbSrvc.knex.raw(`SELECT tenant_panel_constituent_id AS id, tenant_panel_constituent_type AS type FROM tenant_panel_constituents WHERE tenant_panel_id IN (SELECT id FROM tenant_panels WHERE tenant_folder_id IN (${folderDescendants.rows.map((f) => {return '?'}).join(',')}));`, folderDescendants.rows.map((f) => {return f['id']}));

			// eslint-disable-next-line
			const folderAttributeSetPanels = await dbSrvc.knex.raw(`SELECT tenant_panel_id AS id FROM tenant_panels_attribute_sets WHERE tenant_attribute_set_id IN (SELECT id FROM tenant_attribute_sets WHERE tenant_folder_id IN (${folderDescendants.rows.map((f) => {return '?'}).join(',')}));`, folderDescendants.rows.map((f) => {return f['id']}));

			await tenantFolder.destroy();

			cachedFolderArtifacts.shift().forEach((cachedArtifact) => {
				cacheMulti.delAsync(cachedArtifact);
			});

			for(let idx = 0; idx < folderPanelConstituents.rows.length; idx++) {
				const constituent = folderPanelConstituents.rows[idx];
				await ApiService.execute('DataProcessorManager::_updatePanelConstituentsRenderedProcessor', [constituent.id, constituent.type, ctxt.state.tenant['tenant_id']]);
			}

			for(let idx = 0; idx < folderAttributeSetPanels.rows.length; idx++) {
				const panel = folderAttributeSetPanels.rows[idx];
				await ApiService.execute('DataProcessorManager::_updatePanelRenderedProcessor', [panel.id, ctxt.state.tenant['tenant_id']]);
			}

			await cacheMulti.execAsync();
			return null;
		}
		catch(err) {
			throw new PlantWorksMiddlewareError(`${this.name}::_deleteTenantFolder`, err);
		}
	}

	async _getHistoricalData(ctxt) {
		try {
			const cassandra = this.$dependencies.CassandraService;

			const requestedData = ctxt.request.body;
			const requestedDataPoints = Object.keys(requestedData);

			const currentTime = moment();
			currentTime.set({ 'seconds': 0, 'milliseconds': 0 });

			let diffFromFifteenMinuteBoundary = currentTime.minute() % 15;
			if(diffFromFifteenMinuteBoundary >= 8) {
				diffFromFifteenMinuteBoundary = (15 - diffFromFifteenMinuteBoundary);
				currentTime.minute(currentTime.minute() + diffFromFifteenMinuteBoundary);
			}
			else {
				currentTime.minute(currentTime.minute() - diffFromFifteenMinuteBoundary);
			}

			const historicalData = {};
			for(let idx = 0; idx < requestedDataPoints.length; idx++) {
				const dataPoint = requestedDataPoints[idx];
				const dataPointPeriod = requestedData[dataPoint];
				const dataPointTime = currentTime.clone();

				switch (dataPointPeriod) {
					case 'last_one_hour':
						dataPointTime.subtract(1, 'hour');
						break;

					case 'last_two_hours':
						dataPointTime.subtract(2, 'hour');
						break;

					case 'last_four_hours':
						dataPointTime.subtract(4, 'hour');
						break;

					case 'last_eight_hours':
						dataPointTime.subtract(8, 'hour');
						break;

					case 'last_twelve_hours':
						dataPointTime.subtract(12, 'hour');
						break;

					case 'last_twenty_four_hours':
						dataPointTime.subtract(24, 'hour');
						break;

					case 'current_shift':
						// TODO: Once Rahul Khunt's API come online
						break;

					case 'current_day':
						dataPointTime.set({
							'hour': 0,
							'minute': 0,
							'second': 0,
							'millisecond': 0
						});
						break;

					default:
						dataPointTime.subtract(1, 'hour');
						break;
				}

				const dataPointCQL = `SELECT generated_at AS timestamp, value FROM ${ctxt.state.tenant.sub_domain}.panel_${ctxt.params.panel_id.replace(/-/g, '_')} WHERE internal_tag = ? AND generated_at >= ? ORDER BY generated_at ASC`;
				const dataPointData = await cassandra.execute(dataPointCQL, [dataPoint, dataPointTime.format()], { 'prepare': true });

				if(!historicalData[dataPointPeriod]) historicalData[dataPointPeriod] = {};
				historicalData[dataPointPeriod][dataPoint] = dataPointData.rows.map((data) => {
					return {
						'timestamp': moment(data.timestamp).format(),
						'value': data.value
					};
				});
			}

			return historicalData;
		}
		catch(err) {
			throw new PlantWorksMiddlewareError(`${this.name}::_getHistoricalData`, err);
		}
	}
