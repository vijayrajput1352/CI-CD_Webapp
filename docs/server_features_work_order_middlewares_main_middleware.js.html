	async _getTreeNodes(ctxt) {
		try {
			const dbSrvc = this.$dependencies.DatabaseService;

			if(ctxt.query.node_type === 'root-folder') {
				let tenantFolders = null;

				// Get the sub-folders
				tenantFolders = await dbSrvc.knex.raw(`SELECT * FROM fn_get_folder_descendants(?) WHERE level = 2`, [ctxt.query.node_id]);
				tenantFolders = tenantFolders.rows.map((folder) => {
					return {
						'id': folder.id,
						'parent': folder.parent_id,
						'text': folder.name,
						'children': true,

						'li_attr': {
							'title': folder.name
						},

						'data': {
							'configsRoute': 'configure.work-order',
							'dataUrl': '/work-order/tree-nodes',
							'intl': true,
							'type': folder.name === 'work_order_feature.folder_names.attribute_sets.name' ? 'attribute-folder' : 'format-folder'
						}
					};
				});

				return tenantFolders;
			}

			if(['attribute-folder', 'format-folder'].includes(ctxt.query.node_type)) {
				// Get the sub-folders
				let tenantFolders = await dbSrvc.knex.raw(`SELECT * FROM fn_get_folder_descendants(?) WHERE level = 2`, [ctxt.query.node_id]);
				tenantFolders = tenantFolders.rows.map((folder) => {
					return {
						'id': folder.id,
						'parent': folder.parent_id,
						'text': folder.name,
						'children': true,

						'li_attr': {
							'title': folder.name
						},

						'data': {
							'configsRoute': 'configure.work-order',
							'dataUrl': '/work-order/tree-nodes',
							'intl': false,
							'type': ctxt.query.node_type
						}
					};
				});

				// Get the formats in this folder
				let tenantFormats = await dbSrvc.knex.raw(`SELECT id, tenant_folder_id AS parent_id, name FROM tenant_work_order_formats WHERE tenant_folder_id  = ?`, [ctxt.query.node_id]);
				tenantFormats = tenantFormats.rows.map((format) => {
					return {
						'id': format.id,
						'parent': format.parent_id,
						'text': format.name,
						'children': true,

						'li_attr': {
							'title': format.name
						},

						'data': {
							'configsRoute': 'configure.work-order',
							'dataUrl': '/work-order/tree-nodes',
							'type': 'format'
						}
					};
				});

				// Get the attribute sets in this folder
				let tenantAttributeSets = await dbSrvc.knex.raw(`SELECT id, tenant_folder_id AS parent_id, name FROM tenant_attribute_sets WHERE tenant_folder_id  = ?`, [ctxt.query.node_id]);
				tenantAttributeSets = tenantAttributeSets.rows.map((attrSet) => {
					return {
						'id': attrSet.id,
						'parent': attrSet.parent_id,
						'text': attrSet.name,
						'children': true,

						'li_attr': {
							'title': attrSet.name
						},

						'data': {
							'configsRoute': 'configure.work-order',
							'dataUrl': '/work-order/tree-nodes',
							'type': 'attribute-set'
						}
					};
				});

				return [...tenantFolders, ...tenantFormats, ...tenantAttributeSets];
			}
			return [];
		}
		catch(err) {
			throw new PlantWorksMiddlewareError(`${this.name}::_getTreeNodes`, err);
		}
	}

	async _approveUpload(ctxt) {
		try {
			const approvedData = ctxt.request.body;
			const currentTime = moment().format();
			const formattedData = [];
			const propTypeObj = {};
			const cassSrvc = this.$dependencies.CassandraService;
			const dbSrvc = this.$dependencies.DatabaseService;
			const allowedStatusTypes = await this.getWorkOrderStatusTypes();
			if(!allowedStatusTypes.includes('not_started')) throw new Error('status value not_started is not allowed');

			const properties = await dbSrvc.knex.raw(`select internal_tag, datatype data_type from tenant_attribute_set_properties where attribute_set_id in (select tenant_attribute_set_id from tenant_work_order_format_attribute_sets where tenant_work_order_format_id = ?)`, [ctxt.params.id]);

			properties.rows.forEach(row=>{
				propTypeObj[row.internal_tag] = row.data_type;
			});

            const uniqueAttribute = await this.getUniqueAttribute(ctxt.params.id);

            // check unique attribute
			if(!uniqueAttribute.id) throw new Error('no unique attribute found');

			let obj = {
				'work_order_id': null,
				'internal_tag': null,
				'data_type': null,
				'inserted_at': null,
				'metadata': null,
				'value': null,

			};

				delete row.notes;
				Object.keys(row).forEach(attr=>{
				obj = {};
				if(attr !== uniqueAttribute.tag && attr !== 'status') {
				obj['work_order_id'] = row[uniqueAttribute.tag];
				obj['internal_tag'] = attr;
				obj['data_type'] = propTypeObj[attr];
				obj['inserted_at'] = currentTime;
				obj['metadata'] = '';
				obj['value'] = row[attr].toString();
				obj['status'] = row['status'];
				formattedData.push(obj);

			for(let i = 0; i < formattedData.length; i++) {
				const dataObj = formattedData[i];

				}

				const queryObj = {};
				const dataObj = approvedData.rows[i];

					queryObj['params'] = [dataObj[uniqueAttribute.tag].toString(), null, null, moment().format(), 'not_started', ttl];
					queries.push(queryObj);
				}

			await cassSrvc.batch(queries, {'prepare': true});

	async _getFormatWorkOrders(ctxt) {
		try {
			const keyspace = ctxt.state.tenant['sub_domain'];
			// const cassSrvc = this.$dependencies.CassandraService;
			const dbSrvc = this.$dependencies.DatabaseService;

			// get unique attribute
			const uniqueAttribute = await this.getUniqueAttribute(ctxt.params.format_id);

			// select all work order id for this format;
			const woFormatTable = `wo_format_${ctxt.params.format_id.replace(/-/g, '_')}`;

			let query =	{
				'size': 10000,
				'query':	{
					'match_all': {}
				}
			};
			const queryResult = await this.getElasticSearchData(woFormatTable, keyspace, query);
			const woTagMap = {};
			const woIdMap = {};
			const workOrderIds = [];
			queryResult.forEach(data=>{
				const key = `${data['work_order_id']}_${data['internal_tag']}`;
				woTagMap[key] = data['value'];
				if(!(data['work_order_id'] in woIdMap)) {
					woIdMap[data['work_order_id']] = true;
					workOrderIds.push(data['work_order_id']);
				}
			});

			// fetch status data for these work orders
			query = {
				'size': 10000,
				'from': 0,
				'query': {
					'bool': {
						'filter': {
							'terms': {
							'work_order_id': workOrderIds
							}
						}
					}
				},
				'sort': [
					{ 'status_update_time': 'desc' }
				]
			};
			let statusData = await this.getElasticSearchData('work_order_status', keyspace, query);

			const uniqueWorkOrderSet = new Set();
			statusData = statusData.filter(data=>{
				if(uniqueWorkOrderSet.has(data['work_order_id'])) return false;
				else {
					uniqueWorkOrderSet.add(data['work_order_id']);
					if(data['status'] === 'not_started' || data['status'] === 'in_progress' || data['status'] === 'discontinued') return true;
					else return false;
				}
			});

			// make finalWorkOrderArr
			statusData.forEach(data=>{
				attrSetProp.forEach(prop=>{
					if(prop !== uniqueAttribute['tag']) {
						const obj = {
						'work_order_id': data['work_order_id'],
						'internal_tag': prop,
						'value': null,
						'status': data['status'],
						'lineId': data['tenant_plant_unit_line_id'],
						'lineConfig': data['work_order_line_config'] ? JSON.parse(data['work_order_line_config']) : null
						};
						finalWorkOrderArr.push(obj);
					}
				});
			});

			for(let i = 0; i < finalWorkOrderArr.length; i++) {
				const workOrderObj = finalWorkOrderArr[i];
				finalWorkOrderArr[i]['value'] = woTagMap[`${workOrderObj['work_order_id']}_${workOrderObj['internal_tag']}`];
			}

			// format finalWorkOrderArr
			const finalWorkOrderObj = {};
			finalWorkOrderArr.forEach(data=>{
				if(!(data['work_order_id'] in finalWorkOrderObj)) finalWorkOrderObj[data['work_order_id']] = {};
				finalWorkOrderObj[data['work_order_id']][data['internal_tag']] = data['value'];
				finalWorkOrderObj[data['work_order_id']]['status'] = data['status'];
				finalWorkOrderObj[data['work_order_id']]['lineId'] = data['lineId'];
				finalWorkOrderObj[data['work_order_id']]['lineConfig'] = data['lineConfig'];
			});

			const dataArr = [],
			headArr = [];
			attrSetProp.forEach(prop=>{
				if(prop !== uniqueAttribute['tag'])	{
					headArr.push(prop);
				}
			});
			Object.keys(finalWorkOrderObj).forEach(workOrderId=>{
				const row = {
					'work_order_id': workOrderId,
					'status': finalWorkOrderObj[workOrderId]['status'],
					'displayStatus': finalWorkOrderObj[workOrderId]['status'].replace(/_/g, ' '),
					'lineId': finalWorkOrderObj[workOrderId]['lineId'],
					'lineConfig': finalWorkOrderObj[workOrderId]['lineConfig'],
					'data': []
				};
				headArr.forEach(attr=>{
					row.data.push(finalWorkOrderObj[workOrderId][attr]);
				});
				dataArr.push(row);
			});

			const finalObj = {
				'head': headArr,
				'rows': dataArr
			};
			return finalObj;
		}
		catch(err) {
			throw new PlantWorksMiddlewareError(`${this.name}::_getFormatWorkOrders`, err);
		}
	}

	async filterMasterData(finalMasterArr, finalFilterArr, dataTypeMap) {

		const promises = require('bluebird');
		const path = require('path');
		const ejs = promises.promisifyAll(require('ejs'));
		const requireFromString = require('require-from-string');

		const tmplPath = path.join(__dirname, './templates/evaluate_expression.ejs');

		const filteredMasterArr = [];
		for(let i = 0; i < finalMasterArr.length; i++) {
			const masterItem = finalMasterArr[i];
			let flg = true;
			for(let j = 0; j < finalFilterArr.length; j++) {
				const filter = finalFilterArr[j];
				let tagValue = masterItem[filter['parameter_name']];
				let filterFlg = false;
				const dataType = dataTypeMap[filter['parameter_name']];
				let exp = '';
				if(!tagValue || !filter['value'] || !dataType) {
					flg = false;
					break;
				}
				if(dataType === 'string') {
					exp = `'${tagValue.toString().toLowerCase()}' ${filter['comparison_operator'] === '=' ? '===' : filter['comparison_operator']}  '${filter['value'].toString().toLowerCase()}'`;
				}
				if(dataType === 'number' || dataType === 'boolean') {
					exp = `${tagValue} ${filter['comparison_operator'] === '=' ? '===' : filter['comparison_operator']}  ${filter['value']}`;
				}
				if(dataType === 'date') {
					filter['value'] = moment(filter['value']).unix();
					tagValue = moment(tagValue).unix();
					exp = `${tagValue} ${filter['comparison_operator'] === '=' ? '===' : filter['comparison_operator']}  ${filter['value']}`;
				}
				if(dataType === 'object') {
					filter['value'] = (typeof filter['value'] === 'string') ? filter['value'] : JSON.stringify(filter['value']);
					tagValue = (typeof tagValue === 'string') ? tagValue : JSON.stringify(tagValue);
					exp = `${tagValue} ${filter['comparison_operator'] === '=' ? '===' : filter['comparison_operator']}  ${filter['value']}`;
				}

				const evaluator = await ejs.renderFileAsync(tmplPath, {
					'expression': exp
				});

				const moduleExports = requireFromString(evaluator);
				filterFlg = moduleExports.evaluate_expression();

				flg = flg && filterFlg;
			}
			if(flg)filteredMasterArr.push(masterItem);
		}

		return filteredMasterArr;
	}

	async getElasticSearchData(table, keyspace, query) {
		const ElasticService = this.$dependencies.ElasticService;
		const queryResult = [];
		let chunk = null;
		let fetchedDataSize = 0;
		do {
			chunk = await ElasticService.search(table, keyspace, query);
			if(chunk && chunk.hits && chunk.hits.hits) {
				// eslint-disable-next-line no-loop-func
				chunk.hits.hits.forEach(data=>{
					if(data._source)queryResult.push(data._source);
				});
			}
			fetchedDataSize += chunk.hits.hits.length;
			query['from'] = fetchedDataSize;
		} while(fetchedDataSize !== chunk.hits.total);

		return queryResult;
	}
